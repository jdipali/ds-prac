import pandas as pd
import json

input_file = "data.csv"
output_file = "csv_horus.json"

df = pd.read_csv(input_file)
records = df.to_dict(orient="records")

horus = {
    "horus_version": "1.0",
    "source": "CSV",
    "records": records
}

with open(output_file, "w") as f:
    json.dump(horus, f, indent=2)

print("CSV converted to HORUS format!")
